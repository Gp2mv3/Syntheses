\section{Question Oestges : phaseurs}
Soit le circuit suivant opérant à $\SI{20}{\kilo\hertz}$ avec $V_o=2.46 \angle\ang{126.87}$ V,
\begin{center}
  	\begin{circuitikz} 
  		\draw
		(0,2.5) to[american controlled current source,l=$2 V_x$] (0,0) 
		(2.5,2.5) -- (0,2.5)
		(5,2.5) to [american voltage source,l_=$12\angle\ang{0}$ V] (2.5,2.5) 
		(2.5,0) to[R,l=$\SI{1}{\ohm}$,-*] (2.5,2.5)
		(5,2.5) to[C,l=-j$\SI{1}{\ohm}$,i=$I_C$,v=$V_x$] (5,0)
		(5,2.5) to[european resistor,*-,l=j$X$] (7.5,2.5) to [R,l=$\SI{1}{\ohm}$,v=$V_o$] (7.5,0) -- (5,0)
		(7.5,2.5) to [short,*-o,l=$B$] (8.5,2.5)
		(7.5,0) to [short,*-o,l=$A$] (8.5,0)
		(0,0) -- (5,0);
	\end{circuitikz}
\end{center}

  
On demande de calculer
\begin{enumerate}
    \item Le courant $I_C$ (amplitude et phase) dans la capacité
    \item La valeur de l’impédance $\imagj X$ et la valeur de la capacité ou inductance correspondante.
    \item La tension (amplitude et phase) aux bornes de la source commandée de courant, ainsi que sa valeur efficace.
    \item La puissance active délivrée par la source de tension.
    \item Le dipôle équivalent de Thévenin aux bornes $A-B$
	\item L'impédance de charge qui maximiserait le transfert de puissance (à la charge)
\end{enumerate}

\begin{solution}
    Pour la résolution du circuit nous utiliserons les notations :
    \begin{center}
  	\begin{circuitikz} 
  		\draw
		(0,2.5) to[american controlled current source,l=$2 V_x$] (0,0) 
		(2.5,2.5) -- (0,2.5)
		(5,2.5) to [american voltage source,l_=$\SI{12\angle\ang{0}}{\volt}$, i = $I_S$] (2.5,2.5) 
		(2.5,0) to[R,l=$\SI{1}{\ohm}$,i = $I_R$, v = $V_R$,-*] (2.5,2.5)
		(5,2.5) to[C,l=$\SI{-1\imagj}{\ohm}$,i=$I_C$,v=$V_x$] (5,0)
		(5,2.5) to[european resistor,*-,l= $\imagj X$,v =$V_L$] (7.5,2.5) to [R,l=$\SI{1}{\ohm}$,i=$I_0$,v=$V_o$] (7.5,0) -- (5,0)
		(7.5,2.5) to [short,*-o,l=$B$] (8.5,2.5)
		(7.5,0) to [short,*-o,l=$A$] (8.5,0)
		(0,0) -- (5,0);
	\end{circuitikz}
\end{center}
Sur base de la loi d'Ohm ($V=ZI$) nous pouvons écrire :
\begin{equation*}
    I_o = \SI{2.46\angle\ang{126.87}}{\volt}
\end{equation*}
\begin{enumerate}
\item En faisant la loi des mailles sur la maille centrale et une équation des noeuds sur le noeud en bas de la résistance de $\SI{1}{\ohm}$ nous pouvons écrire : 
\begin{equation*}
    \left \{
    \begin{array}{rcl}
    V_R + 12 + V_x &=&0 \\
    2 V_x + I_c + I_o &=& \frac{V_R}{1}
    \end{array}
    \right.
    \Leftrightarrow
    \left \{
    \begin{array}{rcl}
    V_R + V_x &=& =-12 \\
    2 V_x  -V_R + \frac{V_x}{-\imagj }  &=&  -2.46\angle\ang{126.87} 
    \end{array}
    \right.
\end{equation*}
\begin{equation*}
\left \{
\begin{array}{rcl}
V_R + V_x &=& =-12 \\
V_x (2- \frac{1}{\imagj }) - V_R &=&  -2.46\angle\ang{126.87}

\end{array}
\right.
\Leftrightarrow
 \left \{
\begin{array}{rcl}
V_x &=& = \SI{3.385\angle\ang{172.157}}{\volt} \\
V_R &=& \SI{8.658\angle\ang{-176.94}}{\volt} 

\end{array}
\right.
\end{equation*}
On peut trouver le courant $I_c$ sur base de la loi d'Ohm ($V=ZI$):
\begin{equation*}
    I_c = \frac{V_x}{-\imagj } = \SI{3.38\angle{-97.843}}{\ampere}
\end{equation*}
\item En faisant une loi des mailles sur la maille de droite nous pouvons écrire :
\begin{equation*}
    V_L = V_x - V_o = \SI{2.4068\angle\ang{-141.261}}{\volt}
\end{equation*}
On peut trouver $\imagj X$ sur base de la loi d'Ohm ($V = ZI$) :
\begin{equation*}
    \imagj X = \frac{V_L}{I_o}=0.978\angle\ang{91.869} = - 0.032 + 0.978\imagj  \simeq 1\imagj  \Leftrightarrow X = 1 \rightarrow L = \frac{1}{\omega} =\frac{1}{2\pi \cdot 20000} = \SI{7.957}{\micro\henry}
\end{equation*}
\item La tenion au borne de la source commandé vaut $V_R$:
\begin{equation*}
    V_s = \SI{8.658\angle\ang{-176.94}}{\volt} \rightarrow V_{eff} = \frac{V_s}{\sqrt{2}} = \SI{6.12}{\volt}
\end{equation*}
\item En faisant une loi des noeuds sur le noeud de la borne positive de la source de tension nous obtenons : 
\begin{equation*}
    I_s = 2V_x - I_R = 2V_x - V_R = \SI{2.38\angle\ang{35.55}}{\ampere} 
\end{equation*}
En considérant que la tension $V_o$ est valeur de crête nous pouvons écrire\footnote{Si c'était la valeur efficace nous n'aurions pas la facteur 2 au dénominateur} : 
\begin{equation*}
    P = \Re(S) = \frac{VI_s^\ast}{2} =\Re(6 \cdot 2.38\angle\ang{-35.55})= \SI{11.61}{\watt}
\end{equation*}
\item Pour trouver le dipôle équivalent de Thévenin on remarque que $V_0 = \SI{0}{\volt}$.

\begin{center}
  	\begin{circuitikz} 
  		\draw
		(0,2.5) to[american controlled current source,l=$2 V_x$] (0,0) 
		(2.5,2.5) -- (0,2.5)
		(5,2.5) to [american voltage source,l_=$12\angle\ang{0}$ V] (2.5,2.5) 
		(2.5,0) to[R,l=$\SI{1}{\ohm}$,i>=$I_R$,*-*] (2.5,2.5)
		(5,2.5) to[C,l=-j$\SI{1}{\ohm}$,i=$I_C$,v=$V_x$,*-*] (5,0)
		(5,2.5) to[european resistor,*-,l=j$X$] (7.5,2.5)  (7.5,0) -- (5,0)
		(7.5,2.5) -- (8.5,2.5) to [short, i>=$I_N$] (8.5,0)
		(7.5,0) -- (8.5,0)
		(0,0) -- (5,0);
	\end{circuitikz}
\end{center}
Nous pouvons observer que :
\begin{equation*}
    \left \{
    \begin{array}{rcl}
    &2V_x +I_C +I_N =  I_R \\
    &V_x  =  jX \cdot I_N~~\mbox{avec }X=1
    \end{array}
    \right.
    \Leftrightarrow
    \left \{
    \begin{array}{rcl}
    &2V_x +\frac{V_x}{-\imagj} +I_N = - \frac{V_x+12}{1} \\
    &I_N =  \frac{V_x}{\imagj} 
    \end{array}
    \right.
\end{equation*}
\begin{align*}
    &\rightarrow 2 V_x +\frac{V_x+12}{1}+\frac{V_x}{-\imagj}+\frac{V_x}{\imagj} = 0\\ 
    &\Leftrightarrow  3V_x = -12\\
    &\Leftrightarrow V_x = \SI{-4}{\volt}\\
    &\rightarrow I_N = \frac{-4}{\imagj} = 4\imagj = \SI{4\angle\ang{90}}{\ampere}
\end{align*}
\begin{equation*}
    Z_{eq}= \frac{V_{Th}}{I_N} = \frac{2.46\angle\ang{126.87}}{4\imagj}=0.615\angle\ang{36.87}
\end{equation*}
Le dipôle équivalent de Thévenin aux bornes $A-B$ est représenté ci-dessous : 
\begin{center}
        \begin{circuitikz}
        \draw
        (0,0) to [american voltage source,l=$V_{Th}$] (0,2) to [R,l=$Z_{eq}$](3,2)
        (3,0)--(0,0);
        \end{circuitikz}
    \end{center}
\item L'impédance de charge qui maximiserait le transfert de la puissance est $Z_{charge} = Z_{eq}^\ast$. Prouvons l'égalité sur base d'un cas général\footnote{Une démonstration plus rigoureuse se trouve dans le livre de référence à la page 393}: 
\begin{center}
        \begin{circuitikz}
        \draw
        (0,0) to [american voltage source,l=$V$] (0,2) to [european resistor,l=$Z_1 $](3,2) to [european resistor, l=$Z_2$](3,0)--(0,0);
        \end{circuitikz}
    \end{center}
    Avec $Z_1= R1+\imagj X_1$ et $Z_2 = R_2 + \imagj X_2$. La puissance de $Z_2$ est donné par :
\begin{equation*}
    P_{Z_2} = \Re(V_2I^\ast) = \Re( Z_2 I I^\ast) = \Re(Z_2 |I|^2) ~~\mbox{avec}~~I=\frac{V}{\sqrt{(R_1+R_2)^2+(X_1+X_2)^2}}
\end{equation*}
\begin{equation*}
    \Leftrightarrow P_{Z_2} =\Re(Z_2 |I|^2) = R_2 \frac{V^2}{(R_1+R_2)^2+(X_1+X_2)^2}
\end{equation*}
Pour maximiser sa puissance nous allons voir pour quelles valeurs sa dérivée par rapport à $X_2$ et $R_2$ s'annule :
\begin{align*}
    \frac{\partial P_{Z_2}}{\partial X_2} &= -\frac{2R_2V^2\left(X_2+X_1\right)}{\left(\left(X_2+X_1\right)^2+\left(R_2+R_1\right)^2\right)^2} =0\\
    &\Rightarrow X1 = -X2\\
    \frac{\partial P_{Z_2}}{\partial R_2} &= 
    -\frac{V^2\left(R_2^2-\left(X_2+X_1\right)^2-R_1^2\right)}{\left(\left(R_2+R_1\right)^2+\left(X_2+X_1\right)^2\right)^2}=0\\
    &\Rightarrow R_1 = R_2 ~~\mbox{et}~~X_1 = -X_2 
\end{align*}

On peut voir que cette condition amène un maximum en regardant le signe de la dérivée seconde : 
\begin{align*}
    \frac{\partial^2 P_{Z_2}}{\partial X_2^2}&= \frac{2R_2V^2\left(3\left(X_2+X_1\right)^2-\left(R_2+R_1\right)^2\right)}{\left(\left(X_2+X_1\right)^2+\left(R_2+R_1\right)^2\right)^3}\rightarrow - \frac{2R_2V^2(R1+R2)^2}{\left(\left(X_2+X_1\right)^2+\left(R_2+R_1\right)^2\right)^3}\\
    \frac{\partial^2 P_{Z_2}}{\partial R_2^2}&=\frac{4V^2\left(R_2+R_1\right)\left(R_2^2-\left(X_2+X_1\right)^2-R_1^2\right)-2V^2R_2\left(\left(R_2+R_2+X_1\right)^2\right)}{\left(\left(R_2+R_1\right)^2+\left(X_2+X_1\right)^2\right)^3} \rightarrow-\frac{2V^2R_2}{(R1+R2)^4}
\end{align*}
Dont les signe sont négatif lorsque $X_1=-X_2$ et $R_1=R_2$. Cela indique que la puissance est maximale lorsque $Z_1 = R_1+\imagj X_1 = R_2 - \imagj X_2 = Z_2^\ast$
\end{enumerate}
\end{solution}

\section{Question Oestges : Bode et quadripôles}
On considère le circuit suivant
\begin{center}
	\begin{circuitikz} 
		\draw
  		(2.5,2.5) to [european resistor,l^=$Z_1$,i=$I_i$]  (5,2.5)
 		(5,2.5) to[european resistor,l=$Z_2$] (5,0)
 		(7.5,2.5) to[european resistor,-*,l_=$Z_3$,i=$I_o$] (5,2.5)
  		(7,2.5) to [short,-o] (8,2.5)
 		(5,0) to [short,-o] (8,0)
 		(5,0) to [short,-o] (2,0)
 		(2.5,2.5) to [short,-o] (2,2.5)
 		(2,2.5) to [open,v=$V_i$] (2,0)
		(8,2.5) to [open,v^=$V_o$] (8,0);
	\end{circuitikz}
\end{center}
Le quadripôle de ce circuit est représenté par la matrice $Z$ suivante à la fréquence de $\SI{26.5}{\kilo\hertz}$
\[ \begin{bmatrix} 6-2\imagj  & 4-6\imagj  \\ 4-6\imagj  & 7+2\imagj  \end{bmatrix} \]
On demande de
\begin{enumerate}
    \item Déterminer l’expression analytique (en fonction de $Z_1$, $Z_2$ et $Z_3$) de la matrice $Z$ du quadripôle représenté.
    \item Calculer les valeurs complexes de $Z_1$, $Z_2$ et $Z_3$.
    \item Sachant que $Z_1$ et $Z_3$ sont des impédances de type R-L série ($R_1$, $L_1$ et $R_3$, $L_3$) et que $Z_2$ est une impédance de type R-C série ($R_2$, $C_2$) calculer les différents composants.
    \item Réécrire la matrice $Z$ du quadripôle en fonction des différent composants trouvés ci-dessus ainsi que de la pulsation $\omega$.
    \item Tracer le diagramme de Bode du gain $A_\text{vf}$ pour les valeurs calculées des éléments si on branche en sortie du quadripôle une résistance très grande (supposée infinie). De quel type de filtre s’agit-il ?
    \item Calculer l'impédance d’entrée du circuit si on branche en sortie du quadripôle une résistance de $\SI{5}{\ohm}$.
\end{enumerate}

\begin{solution}
\begin{enumerate}
    \item La matrice $Z$ est calculée grâce à : 
    \begin{equation*}
        \begin{bmatrix}
        V_i \\
        V_o
        \end{bmatrix} = 
\begin{bmatrix}
z_i & z_r \\
z_f & z_o
\end{bmatrix}
\begin{bmatrix}
I_i \\
I_o
\end{bmatrix}
\end{equation*}
En annulant le courant $I_o$ le circuit devient :
\begin{center}
	\begin{circuitikz} 
		\draw
  		(2.5,2.5) to [european resistor,l^=$Z_1$,i=$I_i$]  (5,2.5)
 		(5,2.5) to[european resistor,*-,l=$Z_2$] (5,0)
 		(7.5,2.5) to (5,2.5)
  		(7,2.5) to [short,-o] (8,2.5)
 		(5,0) to [short,-o] (8,0)
 		(5,0) to [short,-o] (2,0)
 		(2.5,2.5) to [short,-o] (2,2.5)
 		(2,2.5) to [open,v=$V_i$] (2,0)
		(8,2.5) to [open,v^=$V_o$] (8,0);
	\end{circuitikz}
\end{center}
Sur base de la loi d'Ohm ($Z=RI$) nous pouvons écrire :
\begin{equation*}
\left \{
    \begin{array}{rcl}
    V_i &=& (Z_1+ Z_2)I_i \\
    V_o &=&  Z_2 I_o
\end{array}
\right.
\Leftrightarrow 
\left \{
    \begin{array}{rcl}
    z_f &=& Z_1+ Z_2 [\Omega]\\
    z_i &=&  Z_2 [\Omega]
\end{array}
\right.
\end{equation*}
En annulant le courant $I_i$ le circuit devient : 
\begin{center}
	\begin{circuitikz} 
		\draw
  		(2.5,2.5) to   (5,2.5)
 		(5,2.5) to[european resistor,l=$Z_2$] (5,0)
 		(7.5,2.5) to[european resistor,-*,l_=$Z_3$,i=$I_o$] (5,2.5)
  		(7,2.5) to [short,-o] (8,2.5)
 		(5,0) to [short,-o] (8,0)
 		(5,0) to [short,-o] (2,0)
 		(2.5,2.5) to [short,-o] (2,2.5)
 		(2,2.5) to [open,v=$V_i$] (2,0)
		(8,2.5) to [open,v^=$V_o$] (8,0);
	\end{circuitikz}
\end{center}

Sur base de la loi d'Ohm $Z=RI$ nous pouvons écrire :
\begin{equation*}
\left \{
    \begin{array}{rcl}
    V_i &=&  Z_2 I_o \\
    V_o &=&  (Z_2+Z_3) Io
\end{array}
\right.
\Leftrightarrow 
\left \{
    \begin{array}{rcl}
    z_r &=&  Z_2 [\Omega]\\
    z_0 &=&  Z_2+Z_3 [\Omega]
\end{array}
\right.
\end{equation*}
La matrice Z est donc :
\begin{equation*}
    Z = \begin{bmatrix} Z_1 + Z_2 & Z_2 \\ Z_2 & Z_2 + Z_3 \end{bmatrix}
\end{equation*}

\item Nous pouvons calculer les valeurs complexes de $Z_1$, $Z_2$ et $Z_3$ avec les valeurs données et la matrice Z calculée précédement :
\begin{equation*}
  \left \{
        \begin{array}{rcl}
    Z_1 +Z_2 &=&  6-2\imagj  \\
    Z_2 &=&  4-6\imagj  \\
    Z_2+Z_3 &=& 7+2\imagj 
\end{array}
\right.
\Leftrightarrow 
\left \{
    \begin{array}{rcl}
    Z_1 &=& 2+4\imagj  [\Omega] \\
    Z_2 &=& 4-6\imagj  [\Omega]\\
    Z_3 &=& 3+8\imagj  [\Omega]
\end{array}
\right.
\end{equation*}
\item Nous pouvons réécrire les relations calculées précédement sous la forme :
\begin{equation*}
\left\{\begin{matrix*}[l]
Z_1 = 2+4\imagj  = R_1+ \imagj \omega L_1 \\
Z_2 = 4-6\imagj  = R_2 + \imagj \frac{1}{\omega C_2} \\
Z_3 = 3+8\imagj  = R_3 +\imagj \omega L_3
\end{matrix*}\right. 
    \Leftrightarrow
    \left\{\begin{matrix*}[l]
    R_1 = 2 [\Omega]\\
    R_2 = 4 [\Omega]\\
    R_3 = 3 [\Omega]\\
    L_1 =  \frac{4}{\omega} = \frac{4}{2*\pi*26500} = 24[\mu H]\\
    C_2 = \frac{1}{6\omega} = \frac{1}{6*2*\pi*26500}=1[\mu F]\\
    L_3 =\frac{8}{\omega} = \frac{8}{2*\pi*26500} = 48[\mu H]\\[1ex]\end{matrix*}\right.
\end{equation*}

\item La matrice $Z$ est la suivante
	\[ Z = \begin{bmatrix}R_1+R_2+\imagj \omega L_1 + \frac{1}{\imagj \omega C_2} & R_2 + \frac{1}{\imagj \omega C_2} \\ R_2 + \frac{1}{\imagj \omega C_2} & R_2+R_3+\imagj \omega L_3 + \frac{1}{\imagj \omega C_2} \end{bmatrix} \]
	
\item Le gain s'écrit :
   \begin{equation*}
       A_{vf} = \frac{A_{vfo}}{1-\frac{z_r}{z_l}A_{vfo}} ~~\mbox{avec} ~~A_{vfo}= \frac{z_f}{z_i}\frac{z_L}{z_L+z_o} = \frac{z_f}{z_i}~~\mbox{car} ~~z_L \simeq \infty
   \end{equation*}
   \begin{equation*}
       A_{vf}= \frac{\frac{z_f}{z_i}}{1-\frac{z_r}{z_L}\frac{z_f}{z_i}}= \frac{z_f}{z_i}=\frac{R_2+\frac{1}{\imagj \omega C_2}}{R_1+R_2+\imagj \omega L_1+\frac{1}{\imagj \omega C_2}} = \frac{1+R_2 C_2 \imagj  \omega}{1+(R_1+R_2)\imagj \omega C_2 + (\imagj \omega)^2 L_1 C_2}  
       \end{equation*}
       La fonction de transfert peut être réécrite sous la forme : 
       \begin{equation*}
           H(\imagj \omega) = \frac{1 + \imagj \frac{\omega}{\omega_0}}{1+2\xi \imagj \frac{\omega}{\omega_1}+ (\imagj \frac{\omega}{\omega_1})^2}~~
           \mbox{avec}~~
           \left\{\begin{matrix*}[l]
\omega_0 = \frac{1}{R_2C_2} =2.5*10^{5} [\mbox{rad/s}]\\
\omega_1 =\frac{1}{\sqrt{L_1C_1}} = 204124 [\mbox{rad/s}] \\
\xi = \frac{\omega_1}{2}(R_1+R_2)C_2 = 0.6124
\end{matrix*}\right.
\end{equation*}

    \item L'impédance d'entrée s'écrit :
    \begin{equation*}
        Z_{in} = z_i (1+\frac{z_r}{z_i}A_{if,o})~~ \mbox{avec}~~A_{if,o} = -\frac{z_f}{z_o+z_L}
    \end{equation*}
