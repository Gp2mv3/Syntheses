upon event <Init> do
    delivered := $\emptyset$

upon event <pp2pDeliver | dst, m> do
    trigger <sp2pSend | dst, m>

upon event <sp2pDeliver | src, m> do
    if m $\notin$ delivered then
        delivered := delivered $\cup$ {m}
        trigger <pp2pDeliver | src, m>
