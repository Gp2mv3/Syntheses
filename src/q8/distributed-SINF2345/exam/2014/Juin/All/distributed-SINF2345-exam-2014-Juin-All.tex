\section{Shared registers (3pts)}

\begin{framed}
    Define precisely the following specifications:

    \begin{enumerate}
        \item Regular register property
        \item Sequential consistency
        \item Linearizability
    \end{enumerate}

    Then give example executions to show:

    \begin{enumerate}
        \item A regular register that is not sequentially consistent.
        \item A sequentially consistent register that is not regular.
        \item A sequentially consistent register that is not linearizable.
    \end{enumerate}
\end{framed}
