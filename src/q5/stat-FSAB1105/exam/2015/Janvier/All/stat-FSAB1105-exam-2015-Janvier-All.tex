\section{}

Soit $X$, une variable aléatoire dont la fonction de distribution est donnée par 
\[ F(x,\mu,\beta)=\e^{-\e^{(\mu-x)/\beta}} \qquad x \in \Re \]
où $\mu\in\Re$ et $\beta>0$ sont des paramètres inconnus.

\begin{enumerate}[label=(\alph*)]
 \item Calculer la fonction génératrice de moments de $X$.
 \item Calculer l'espérance et la variance de $X$.
\end{enumerate}

\begin{solution}
 
 On remarque tout d'abord qu'il s'agit de la fonction de distribution cumulative de $X$ car $F$ est en majuscule, $F(x=-\infty,\mu,\beta)=0$ et $F(x=+\infty,\mu,\beta)=1$.\\
 
 On calcule donc la fonction de densité de $X$,
 \[ f(x,\mu,\beta)=\fpart{F}{x}(x,\mu,\beta)=\frac{1}{\beta}\e^{(\mu-x)/\beta}\e^{-\e^{(\mu-x)/\beta}} \]
 
 \begin{enumerate}[label=(\alph*)]
  \item 
   \[ m_x(t,\mu,\beta)=\mathrm{E}[\e^{tx}]=\int_{-\infty}^{+\infty}\e^{tx}\frac{1}{\beta}\e^{(\mu-x)/\beta}\e^{-\e^{(\mu-x)/\beta}} \:\dif x=-\int_{+\infty}^0 \e^{\mu t} y^{-\beta t} \e^{-y} \:\dif y = \e^{\mu t} \underbrace{\int^{+\infty}_0 y^{-\beta t} \e^{-y} \:\dif y}_{\Gamma(1-\beta t)} \]
   où $\Gamma(x)$ représente la fonction Gamma\footnote{Cette fonction est à connaître/noter dans le formulaire.} évaluée en $x$.
  
  \item On calcule l'espérance et la variance à partir de $m_x(t)$,
  \begin{align*}
   \mathrm{E}[X]=\left.\fpart{m}{t}(t,\mu,\beta)\right|_{t=0} 
   & = \left. \mu \e^{\mu t}\Gamma(1-\beta t)\right|_{t=0} + \left.\e^{\mu t}\fpart{}{t}\left(\Gamma(1-\beta t)\right)\right|_{t=0} \\
   & = \mu - \beta \Gamma'(1) \\
   & = \mu + \beta \gamma
  \end{align*}
  avec $\gamma\overset{\Delta}{=}-\Gamma'(1)$, la constante d'Euler-Mascheroni\footnote{Cette constante n'est par contre pas à connaître actuellement.}.
  
  \begin{align*}
   \mathrm{E}[X^2]=\left.\ffpart{m}{t}(t,\mu,\beta)\right|_{t=0} 
   & = \mu^2 + 2\mu\beta\gamma + \left.\e^{\mu t}\ffpart{}{t}\left(\Gamma(1-\beta t)\right)\right|_{t=0} \\
   & = \mu^2 + 2\mu\beta\gamma + \beta^2\Gamma''(1)
  \end{align*}
  
  On trouve ainsi la variance de $X$,
  \[ \mathrm{V}[X]=\mathrm{E}[X^2]-\left(\mathrm{E}[X]\right)^2=\mu^2 + 2\mu\beta\gamma + \beta^2\Gamma''(1)-\left(\mu + \beta \gamma\right)^2 = \beta^2\left(\Gamma''(1)-\gamma^2\right) \]

 \end{enumerate}
 
\end{solution}
